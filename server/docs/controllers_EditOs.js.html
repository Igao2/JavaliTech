     * @param {object} res "response"
     */
    editOsQuery(req, res) {
        const userId = req.userId.id;


        try {
            function editOsResultQuery() {
                return new Promise((resolve, reject) => {
                    const error500 = {
                        erro: true,

                        erroDetails: {
                            mensagem: "Ouve um erro em nosso sistema. Tente novamente mais tarde."
